void setup() {

  bluetooth.begin(9600);

}

void loop() {

  if (bluetooth.available()) 
    input = bluetooth.read();
 

  if (input == 's' && !flag){

      shuntvoltage = 0;

      current_mA = 0;
      loadvoltage = 0;
      whole_mA = 0;
      power_mW = 0;
  }
  
      if (flag == 1) {
        current_mA = ina219.getCurrent_mA();
        if (current_mA > 0)
          whole_mA += current_mA;
          lcd.setCursor(0,0);
          lcd.print("Current:");
          lcd.print(current_mA);
          lcd.setCursor(0,1);
          lcd.print("Whole:");
          lcd.print(whole_mA);
        
        delay(500);
      }
    
     if(flag && input == 'e'){

      while (input != '/'){
        input = bluetooth.read();

        id += input;
        input = bluetooth.read();

      input = bluetooth.read();

        startwith += input;
        input = bluetooth.read();
      }
      input = bluetooth.read();

        endwith += input;

     Serial.println("Exercise");
